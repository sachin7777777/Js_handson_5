                    let char = s.charAt(i);

                        count[char]++;
                    } else {
                        count[char] = 1;
                    }
                }
                let result = "";
                let keys = Object.keys(count).sort();
                for (let i = 0; i < keys.length; i++) {
                    if (count[keys[i]] > 1) {
                        result += keys[i] + count[keys[i]];
                    }
                }
            
                return result;
};
